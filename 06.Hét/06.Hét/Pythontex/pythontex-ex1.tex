\begin{itemize}
\item Első példa, a forráskód:
\begin{verbatim}
	\begin{pycode}
	print("Python says ``Hello!''")
	\end{pycode}
\end{verbatim}
A megjelenő szöveg:

\begin{pycode}
print("Python says ``Hello!''")
\end{pycode}

\item Második példa, a forráskód (math módban):

\verb!$8 \times 256 = \py{8*256}$!

A megjelenő szöveg:

$8 \times 256 = \py{8*256}$
\end{itemize}
